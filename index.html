Loading from a .svg file

<!-- create an empty svg element on the page -->
<svg id="scene"></svg>

<script type="text/javascript">
    // 'select' the new svg element
    var scene = Snap("#scene");

    // load the image into a javascript object
    Snap.load(
        // the image source
        "images/one45-background.svg",

        // callback function to be run once the image is loaded
        onLoadCallback
    );

    function onLoadCallback ( image ) {
        // after the image is loaded, inject it into #scene
        scene.append( image );
    }
</script>
